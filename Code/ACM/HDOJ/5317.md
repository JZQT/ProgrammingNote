# [HDOJ-5317 RGCDQ](http://acm.hdu.edu.cn/showproblem.php?pid=5317)

## 解题报告

预处理出每一个数`x`的`F(x)`值。

`F(x)`的取值范围为`[1, 7]`。

开个二维数组`sum[N][7]`求`F(x)`取值为`[1, 7]`的前缀和。

`O(1)`查询区间`F(x)`不同取值的个数，根据情况输出答案。

## 代码

```cpp
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>
#include <math.h>
#include <algorithm>
#include <vector>
#include <queue>
#include <stack>

using namespace std;

const int N = 1e6 + 11;

int sum[N][7];
char factor[N];
int primes[N], pn;
bool isprime[N];

void Init()
{
    // init
    pn = 0;
    memset(isprime, true, sizeof(isprime));
    isprime[0] = isprime[1] = false;
    memset(sum, 0, sizeof(sum));
    memset(factor, 1, sizeof(factor));
    // get primes[]
    for (int i=0; i<N; ++i) {
        if (isprime[i]) primes[pn++] = i;
        for (int j=0; j<pn; ++j) {
            if (i * primes[j] >= N) break;
            isprime[i * primes[j]] = false;
            if (i % primes[j] == 0) break;
        }
    }
    // get factor[]
    queue<int> q;
    for (int i=0; i<pn; ++i) {
        q.push(primes[i]);
    }
    while (!q.empty()) {
        int a = q.front();
        q.pop();
        for (int i=0; i<pn; ++i) {
            int b = a * primes[i];
            if (b >= N) break;
            if (a % primes[i] == 0) {
                if (factor[b] < factor[a]) {
                    factor[b] = factor[a];
                    q.push(b);
                }
            } else {
                if (factor[b] < factor[a] + 1) {
                    factor[b] = factor[a] + 1;
                    q.push(b);
                }
            }
        }
    }
    // get sum[][]
    for (int i=2; i<N; ++i) {
        sum[i][factor[i]-1]++;
        for (int j=0; j<7; ++j) {
            sum[i][j] += sum[i-1][j];
        }
    }
    return;
}

int main()
{
    int t;
    int l, r;
    int ans[7];
    scanf("%d", &t);
    Init();
    while (t--) {
        scanf("%d %d", &l, &r);
        for (int i=0; i<7; ++i) {
            ans[i] = sum[r][i] - sum[l-1][i];
        }
        if (ans[6] > 1) {
            puts("7");
        } else if (ans[5] > 1) {
            puts("6");
        } else if (ans[4] > 1) {
            puts("5");
        } else if (ans[3] > 1) {
            puts("4");
        } else if (ans[2] + ans[5] > 1) {
            puts("3");
        } else if (ans[1] + ans[3] + ans[5] > 1) {
            puts("2");
        } else {
            puts("1");
        }
    }
    return 0;
}
```
